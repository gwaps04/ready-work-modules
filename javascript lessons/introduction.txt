    <!-- JavaScript Logic -->
    <script>
        // --- Architecture Section Logic ---
        function toggleArchitecture(id) {
            const desc = document.getElementById(`desc-${id}`);
            const isHidden = desc.classList.contains('hidden-content');
            
            // Hide all first
            ['react', 'bootstrap', 'node'].forEach(key => {
                document.getElementById(`desc-${key}`).classList.add('hidden-content');
            });

            if (isHidden) {
                desc.classList.remove('hidden-content');
            }
        }

        // --- Frontend Lab Logic ---
        const demoState = {
            btn: false,
            shadow: false,
            round: false
        };

        function toggleDemo(type) {
            demoState[type] = !demoState[type];
            updateDemoBox();
            updateButtons(type);
        }

        function updateDemoBox() {
            const box = document.getElementById('demo-box');
            const codeDisplay = document.getElementById('code-display');
            
            // Base classes
            let classes = ['bg-white', 'p-6', 'border', 'border-slate-300', 'inline-block', 'transition-all', 'duration-500', 'ease-in-out'];
            let displayClasses = [];

            if (demoState.btn) {
                classes = classes.filter(c => c !== 'bg-white' && c !== 'border' && c !== 'border-slate-300');
                classes.push('bg-blue-600', 'text-white', 'border-transparent');
                displayClasses.push('btn-primary');
            } else {
                displayClasses.push('btn-light');
            }

            if (demoState.shadow) {
                classes.push('shadow-xl', 'transform', '-translate-y-2');
                displayClasses.push('shadow-lg');
            }

            if (demoState.round) {
                classes.push('rounded-full', 'px-10');
                displayClasses.push('rounded-pill');
            } else {
                // Remove rounded-full if not active, keep basic rounded?
                // For simplicity, we just toggle the specific classes
            }

            box.className = classes.join(' ');
            codeDisplay.textContent = displayClasses.join(' ');
        }

        function updateButtons(type) {
            const btnId = type === 'btn' ? 'toggle-btn' : type === 'shadow' ? 'toggle-shadow' : 'toggle-round';
            const btn = document.getElementById(btnId);
            
            if (demoState[type]) {
                btn.textContent = "Remove";
                btn.className = "px-3 py-1 bg-blue-100 text-blue-700 rounded text-sm font-medium hover:bg-blue-200 transition";
            } else {
                btn.textContent = "Apply";
                btn.className = "px-3 py-1 bg-slate-200 rounded text-sm font-medium hover:bg-slate-300 transition";
            }
        }

        // --- Backend Simulator Logic ---
        function runSimulation() {
            const btn = document.getElementById('sim-btn');
            const log = document.getElementById('sim-log');
            const step1 = document.getElementById('step-1');
            const step2 = document.getElementById('step-2');
            const step3 = document.getElementById('step-3');

            if (btn.disabled) return;
            btn.disabled = true;
            btn.classList.add('opacity-50', 'cursor-not-allowed');
            log.classList.remove('hidden');
            log.innerHTML = '';

            // Reset Styles
            [step1, step2, step3].forEach(el => el.classList.remove('step-active', 'opacity-100'));
            [step2, step3].forEach(el => el.classList.add('opacity-50'));

            // Step 1: React Request
            addLog("React: Sending request GET /api/users...");
            step1.classList.add('step-active');
            
            setTimeout(() => {
                // Step 2: Node Processing
                addLog("Node: Request received. querying database...");
                step1.classList.remove('step-active');
                step2.classList.remove('opacity-50');
                step2.classList.add('step-active', 'opacity-100');

                setTimeout(() => {
                    // Step 3: Database
                    addLog("Database: Found 42 users. Returning data...");
                    step2.classList.remove('step-active');
                    step3.classList.remove('opacity-50');
                    step3.classList.add('step-active', 'opacity-100');

                    setTimeout(() => {
                        // Complete
                        addLog("Node: Sending JSON response to React.");
                        addLog("React: Data received! Updating UI.");
                        step3.classList.remove('step-active');
                        step1.classList.add('step-active'); // Back to client
                        
                        setTimeout(() => {
                            step1.classList.remove('step-active');
                            btn.disabled = false;
                            btn.classList.remove('opacity-50', 'cursor-not-allowed');
                            btn.querySelector('span').textContent = "üîÑ Run Again";
                        }, 1000);
                    }, 1500);
                }, 1500);
            }, 1500);
        }

        function addLog(msg) {
            const log = document.getElementById('sim-log');
            const entry = document.createElement('div');
            entry.textContent = `> ${msg}`;
            entry.className = "mb-1 border-b border-slate-800 pb-1";
            log.appendChild(entry);
            log.scrollTop = log.scrollHeight;
        }

        // --- Chart & Project Logic ---
        const projectData = [
            { label: 'Portfolio', data: [80, 50, 90, 60], color: 'rgba(59, 130, 246, 0.5)', borderColor: 'rgb(59, 130, 246)', desc: "Best for immediate job hunting value." },
            { label: 'Task Pulse', data: [40, 90, 70, 50], color: 'rgba(99, 102, 241, 0.5)', borderColor: 'rgb(99, 102, 241)', desc: "Best for learning Logic & State Management." },
            { label: 'Dev-Gram', data: [90, 40, 50, 90], color: 'rgba(236, 72, 153, 0.5)', borderColor: 'rgb(236, 72, 153)', desc: "Best for Fun & CSS Grid practice." }
        ];

        let myRadarChart;

        function initChart() {
            const ctx = document.getElementById('projectChart').getContext('2d');
            myRadarChart = new Chart(ctx, {
                type: 'radar',
                data: {
                    labels: ['Visuals', 'Logic/Backend', 'Practical Utility', 'Fun Factor'],
                    datasets: projectData.map(p => ({
                        label: p.label,
                        data: p.data,
                        fill: true,
                        backgroundColor: p.color,
                        borderColor: p.borderColor,
                        pointBackgroundColor: p.borderColor,
                        pointBorderColor: '#fff',
                        pointHoverBackgroundColor: '#fff',
                        pointHoverBorderColor: p.borderColor
                    }))
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    elements: {
                        line: { borderWidth: 3 }
                    },
                    scales: {
                        r: {
                            angleLines: { display: true },
                            suggestedMin: 0,
                            suggestedMax: 100,
                            ticks: { display: false } // Hide numbers for cleaner look
                        }
                    },
                    plugins: {
                        legend: { position: 'bottom' }
                    }
                }
            });
        }

        function selectProject(index) {
            const p = projectData[index];
            const rec = document.getElementById('recommendation-text');
            rec.textContent = `Selection: ${p.label} - ${p.desc}`;
            rec.className = "mt-6 text-center font-bold text-white px-6 py-3 rounded-full transition-colors";
            rec.style.backgroundColor = p.borderColor;

            // Highlight in chart (hide others briefly or just update dataset focus)
            // For simplicity, we just update the text below the chart
        }

        function scrollToSection(id) {
            document.getElementById(id).scrollIntoView({ behavior: 'smooth' });
        }

        // Initialize
        window.addEventListener('load', () => {
            initChart();
        });

    </script>
